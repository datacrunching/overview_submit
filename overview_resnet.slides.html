from fastai.callbacks.hooks import *
def hooked_backward(wire=y):
    with hook_output(m[0]) as hook_a: 
        with hook_output(m[0], grad=True) as hook_g:
            preds = m(xb)
            preds[0,int(wire)].backward()
    return hook_a,hook_g
